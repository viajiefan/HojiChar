120class JSONLoader(Filter):
121    """
122    テキストを Json として解釈し, `key` で指定した要素を文字列として
123    doument に格納します.デフォルトの `key` は 'text' です.
124
125    Json の読み込み, あるいは `key` の読み込みに失敗した際には例外を送出します.
126    これらを無視する場合は, `ignore=True` にします. その際, 読み込みに失敗
127    したドキュメントは破棄されます.
128    """
129
130    def __init__(self, key: str = "text", ignore: bool = False, *args: Any, **kwargs: Any) -> None:
131        super().__init__(*args, **kwargs)
132        self.key = key
133        self.ignore = ignore
134
135    def apply(self, document: Document) -> Document:
136        """
137        >>> JSONLoader()( '{"text": "hello, world", "words": 2}' )
138        'hello, world'
139
140        >>> JSONLoader()( '{"text": hello, world ....' ) # Broken JSON
141        Traceback (most recent call last):
142            ...
143        json.decoder.JSONDecodeError: Expecting value: line 1 column 10 (char 9)
144
145        >>> JSONLoader()( '{"words": 2}' )
146        Traceback (most recent call last):
147            ...
148        KeyError: 'text'
149
150        >>> JSONLoader(ignore=True).apply(Document('{"text": hello, world ....' )).is_rejected
151        True
152        """
153        try:
154            data = json.loads(document.text)
155            document.text = str(data[self.key])
156        except Exception as e:
157            if self.ignore:
158                document.is_rejected = True
159                return document
160            else:
161                raise e
162
163        return document

166class JSONDumper(Filter):
167    """
168    Document.text の文字列を json に変換します.
169    必要に応じ Document のメタデータを付与します. これはドキュメントの破棄事由が含まれ、偽陽性の分析に有効です。
170    デフォルトで `skip_rejected` が `False` にセットされており、Document の破棄フラグにかかわらず
171    処理されます。
172    """
173
174    def __init__(
175        self,
176        dump_reason: bool = False,
177        p: float = 1,
178        skip_rejected: bool = False,
179        *args: Any,
180        **kwargs: Any,
181    ) -> None:
182        """
183        Args:
184            dump_reason (bool, optional): `is_rejected`, `reason` エントリをダンプします. Defaults to False.
185            p (float, optional): Apply probability. Defaults to 1.
186            skip_rejected (bool, optional): 破棄済みサンプルを排除しません.
187        """
188        super().__init__(p, skip_rejected, *args, **kwargs)
189        self.dump_reason = dump_reason
190
191    def apply(self, document: Document) -> Document:
192        """
193        >>> JSONDumper()("hojichar")
194        '{"text": "hojichar"}'
195        """
196        text = document.text
197        if self.dump_reason:
198            document.text = json.dumps(
199                {
200                    "text": text,
201                    "is_rejected": document.is_rejected,
202                    "reason": document.reject_reason,
203                },
204                ensure_ascii=False,
205            )
206        else:
207            document.text = json.dumps({"text": text}, ensure_ascii=False)
208        return document

244class NgWordsFilterJa(Filter):
245    """
246    日本語のNGワード(および不適切語)を含む文書を破棄します.
247    `dict_path` で指定したファイルから, キーワードのリストを得ます.
248    ファイルは単語が改行で羅列されたテキストファイルです.
249
250    `ignore_confused` を `True` にすると,
251    偽陽性を軽減するために, カタカナのNGワードは前後にカタカナが無い場合のみNG判定されます.
252    デフォルト値は `False` です.
253    """
254
255    def __init__(
256        self,
257        dict_path: Union[str, PathLike],
258        ignore_confused: bool = False,
259        *args: Any,
260        **kwargs: Any,
261    ) -> None:
262        super().__init__(*args, **kwargs)
263
264        with open(dict_path, encoding="utf-8") as fp:
265            ng_words = fp.read().split("\n")
266        ng_words = [w.strip() for w in ng_words if not len(w) == 0]
267
268        if ignore_confused:
269            words_katakana = []
270            words_not_katakana = []
271            for w in ng_words:
272                if re.fullmatch(r"[ァ-ヴー]+", w):
273                    words_katakana.append(re.escape(w))
274                else:
275                    words_not_katakana.append(re.escape(w))
276            katakana_pat = "|".join(words_katakana)
277            katakana_pat = rf"(?<![ァ-ヴー])({katakana_pat})(?![ァ-ヴー])"
278            pat = "|".join(words_not_katakana) + "|" + katakana_pat
279            self.keyword_pat = re.compile(pat)
280        else:
281            ng_words = [re.escape(w) for w in ng_words]
282            pat = "|".join(ng_words)
283            self.keyword_pat = re.compile(pat)
284
285    def apply(self, doc: Document) -> Document:
286        regex_match = self.keyword_pat.search(doc.text)
287        if regex_match:
288            doc.is_rejected = True
289            self.matched_text = regex_match.group()
290            self.matched_text_neighbor = doc.text[
291                regex_match.start() - 20 : regex_match.end() + 20
292            ]
293
294        return doc
